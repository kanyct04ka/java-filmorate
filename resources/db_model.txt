Film as f
----
id PK int
name varchar(255)
description NULL varchar(2000)
releaseDate NULL date
duration NULL long
rating varchar(5)


Genre as g
----
id PK int
name UNIQUE varchar(30)


FilmGenre as fg
----
id PK int
film_id int FK >- Film.id
genre_id int FK >- Genre.id


User as u
----
id PK int
email varchar(100)
login varchar(100)
name NULL varchar(255)
birthday date


UserRelation as ur
----
id PK int
initiator int FK >- User.id
respondent int FK >- User.id
status varchar(20)


Like as l
----
id PK long
user_id int FK >- User.id
film_id int FK >- Film.id
